    TypeScript webpack打包

    Paradox 6/18/2021 typescript

    # 初始化

    1. 初始化项目 npm init -y
    2. 下载依赖 npm i webpack webpack-cli webpack-dev-server typescript ts-loader clean-webpack-plugin
    3. 创建 webpack 配置文件 webpack.config.js
    const path = require("path");
const HTMLWebpackPlugin = require("clean-webpack-plugin");
const { CleanWebpackPlugin } = require("clean-webpack-plugin");
module.exports = {
  // 指定入口文件
  entry: "./src/index.ts",
  // 指定输出目录
  output: {
    path: path.resolve(__dirname, "dist"),
    filename: "bundle.js",
  },
  // 指定webpack打包时要使用的模块
  module: {
    rules: [
      {
        // test指定规则生效的文件
        test: /\.ts$/,
        // 指定使用的loader
        use: "ts-loader",
        // 要排除的文件
        exclude: /node-modules/,
      },
    ],
  },
  plugins: [
    new CleanWebpackPlugin(),
    new HtmlWebpackPlugin({
      title: "TS测试",
    }),
  ],
};
